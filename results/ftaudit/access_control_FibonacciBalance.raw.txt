pragma solidity ^0.4.22;

contract FibonacciBalance {

    address public fibonacciLibrary;
    // the current fibonacci number to withdraw
    uint public calculatedFibNumber;
    // the starting fibonacci sequence number
    uint public start = 3;
    uint public withdrawalCounter;
    // the fibonancci function selector
    bytes4 constant fibSig = bytes4(sha3("setFibonacci(uint256)"));

    // constructor - loads the contract with ether
    constructor(address _fibonacciLibrary) public payable {
        fibonacciLibrary = _fibonacciLibrary;
    }

    function withdraw() {
        withdrawalCounter += 1;
        // calculate the fibonacci number for the current withdrawal user
        // this sets calculatedFibNumber
        // <yes> <report> ACCESS_CONTROL
        require(fibonacciLibrary.delegatecall(fibSig, withdrawalCounter));
        msg.sender.transfer(calculatedFibNumber * 1 ether);
    }

    // allow users to call fibonacci library functions
    function() public {
        // <yes> <report> ACCESS_CONTROL
        require(fibonacciLibrary.delegatecall(msg.data));
    }
}

// library contract - calculates fibonacci-like numbers;
contract FibonacciLib {
    // initializing the standard fibonacci sequence;
    uint public start;
    uint public calculatedFibNumber;

    // modify the zeroth number in the sequence
    function setStart(uint _start) public {
        start = _start;
    }

    function setFibonacci(uint n) public {
        calculatedFibNumber = fibonacci(n);
    }

    function fibonacci(uint n) internal returns (uint) {
        if (n == 0) return start;
        else if (n == 1) return start + 1;
        else return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

=== MODEL_OUTPUT ===
0. The contract uses delegatecall in a way that could potentially allow unauthorized access to critical functions. Specifically, in the FibonacciBalance contract, the withdraw() function uses delegatecall to call a function in the FibonacciLib contract (setFibonacci). This interaction occurs before transferring funds to the user. If the FibonacciLib contract is malicious or compromised, it could manipulate the calculatedFibNumber variable, leading to incorrect calculations and potentially allowing users to withdraw more funds than they should be entitled to. Additionally, the fallback function in FibonacciBalance allows any user to call delegatecall on the FibonacciLibrary contract, which could be exploited if the FibonacciLibrary contract is malicious or compromised. These issues relate to access control as they involve unauthorized external calls that can alter the contract's state before any checks are made.
